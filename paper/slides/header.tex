% header.tex
% Beamer preamble recreating the LaTeXTemplates.com "Beamer Presentation" by Vel
% (https://www.latextemplates.com/template/beamer-presentation)
% Customised for academic economics conference talks.
%
% Visual signature:
%   - Rounded blue title block with drop shadow
%   - Solid blue footer bar with author / short title / date / slide number
%   - Palatino serif body, Helvetica sans headings
%   - White content background, blue frame titles
%
% Drop this file beside slides.qmd and reference it via
% include-in-header: header.tex in the Quarto YAML.

% ---------------------------------------------------------------
% Packages
% ---------------------------------------------------------------
\usepackage{tikz}
\usetikzlibrary{shadows,calc}
\usepackage{xcolor}

% Fonts (xelatex only): macOS system Palatino and Helvetica Neue,
% with STIX Two Math for unicode-math symbols.
% Why not mathpazo: it does not cooperate with xelatex. Symbols drop out.
\usepackage{fontspec}
\usepackage{unicode-math}
\setmainfont{Palatino}
\setsansfont{Helvetica Neue}
\setmonofont{Menlo}
\setmathfont{STIX Two Math}

% ---------------------------------------------------------------
% Colours: classic econ dark blue
% ---------------------------------------------------------------
\definecolor{econblue}{RGB}{44, 58, 150}
\definecolor{econbluelight}{RGB}{80, 94, 180}
\definecolor{econbluedark}{RGB}{30, 40, 110}
\definecolor{econaccent}{RGB}{232, 119, 34}   % warm accent for highlights

\setbeamercolor{structure}{fg=econblue}
\setbeamercolor{frametitle}{fg=econblue, bg=white}
\setbeamercolor{title}{fg=white, bg=econblue}
\setbeamercolor{subtitle}{fg=white, bg=econblue}
\setbeamercolor{section in toc}{fg=econblue}
\setbeamercolor{subsection in toc}{fg=econblue}
\setbeamercolor{item}{fg=econblue}
\setbeamercolor{enumerate item}{fg=econblue}
\setbeamercolor{itemize item}{fg=econblue}
\setbeamercolor{itemize subitem}{fg=econbluelight}
\setbeamercolor{block title}{bg=econblue, fg=white}
\setbeamercolor{block body}{bg=econblue!6, fg=black}
\setbeamercolor{alerted text}{fg=econaccent}
\setbeamercolor{example text}{fg=econbluedark}

% ---------------------------------------------------------------
% Hyperlinks
% ---------------------------------------------------------------
\hypersetup{
  colorlinks=true,
  linkcolor=econblue,
  urlcolor=econblue,
  citecolor=econblue
}

% ---------------------------------------------------------------
% Title page: rounded blue block with drop shadow
% ---------------------------------------------------------------
\setbeamertemplate{title page}{%
  \vspace{0.4em}
  \begin{center}
    \begin{tikzpicture}
      \node[
        fill=econblue,
        rounded corners=6pt,
        text width=0.92\textwidth,
        inner sep=14pt,
        align=center,
        drop shadow={shadow xshift=2pt, shadow yshift=-2pt, opacity=0.35, fill=black!60}
      ]{%
        \color{white}%
        {\LARGE\bfseries\inserttitle\par}%
        \ifx\insertsubtitle\empty\else
          \vspace{0.35em}%
          {\large\insertsubtitle\par}%
        \fi
      };
    \end{tikzpicture}
  \end{center}
  \vspace{1.2em}
  \begin{center}
    {\large\insertauthor\par}
    \vspace{0.6em}
    {\normalsize\itshape\insertinstitute\par}
    \vspace{0.9em}
    {\normalsize\insertdate\par}
  \end{center}
}

% ---------------------------------------------------------------
% Footer: four-segment solid blue bar
% ---------------------------------------------------------------
\setbeamertemplate{footline}{%
  \leavevmode%
  \hbox{%
    \begin{beamercolorbox}[wd=0.30\paperwidth, ht=3.0ex, dp=1.4ex, left, leftskip=1em]{author in head/foot}%
      \usebeamerfont{author in head/foot}\insertshortauthor%
    \end{beamercolorbox}%
    \begin{beamercolorbox}[wd=0.35\paperwidth, ht=3.0ex, dp=1.4ex, center]{title in head/foot}%
      \usebeamerfont{title in head/foot}\insertshorttitle%
    \end{beamercolorbox}%
    \begin{beamercolorbox}[wd=0.20\paperwidth, ht=3.0ex, dp=1.4ex, center]{date in head/foot}%
      \usebeamerfont{date in head/foot}\insertshortdate%
    \end{beamercolorbox}%
    \begin{beamercolorbox}[wd=0.15\paperwidth, ht=3.0ex, dp=1.4ex, right, rightskip=1em]{page number in head/foot}%
      \usebeamerfont{page number in head/foot}\insertframenumber\,/\,\inserttotalframenumber%
    \end{beamercolorbox}%
  }%
}

\setbeamercolor{author in head/foot}{fg=white, bg=econblue}
\setbeamercolor{title in head/foot}{fg=white, bg=econbluelight}
\setbeamercolor{date in head/foot}{fg=white, bg=econblue}
\setbeamercolor{page number in head/foot}{fg=white, bg=econblue}

\setbeamerfont{author in head/foot}{size=\footnotesize, series=\bfseries}
\setbeamerfont{title in head/foot}{size=\footnotesize}
\setbeamerfont{date in head/foot}{size=\footnotesize}
\setbeamerfont{page number in head/foot}{size=\footnotesize, series=\bfseries}

% ---------------------------------------------------------------
% Frame title: large blue, left aligned
% ---------------------------------------------------------------
\setbeamerfont{frametitle}{size=\Large, series=\bfseries}
\setbeamertemplate{frametitle}[default][left]

% ---------------------------------------------------------------
% Blocks (definition / theorem / example look-alike)
% ---------------------------------------------------------------
\setbeamertemplate{blocks}[rounded][shadow=true]

% ---------------------------------------------------------------
% Navigation symbols: keep the small arrow set at bottom-right
% (as in the LaTeXTemplates original). Uncomment to hide.
% ---------------------------------------------------------------
% \setbeamertemplate{navigation symbols}{}

% ---------------------------------------------------------------
% Graphics path: figures directory beside slides.qmd
% ---------------------------------------------------------------
\graphicspath{{figures/}}

% ---------------------------------------------------------------
% Tighter bullet spacing
% ---------------------------------------------------------------
\setlength{\leftmargini}{1.2em}
\setbeamertemplate{itemize item}[circle]
\setbeamertemplate{itemize subitem}[circle]
\setbeamertemplate{itemize subsubitem}[circle]

% ---------------------------------------------------------------
% Caption without 'Figure N:' prefix (slides look cleaner)
% ---------------------------------------------------------------
\setbeamertemplate{caption}{\tiny\insertcaption}
\setbeamertemplate{caption label separator}{: }

% ---------------------------------------------------------------
% Footnote: clean left-aligned layout, no odd hanging indent
% ---------------------------------------------------------------
\setbeamertemplate{footnote}{%
  \parindent 0em\noindent%
  \raggedright
  \usebeamerfont{footnote}\hbox to 1.2em{\hfil\insertfootnotemark\,}\insertfootnotetext\par%
}
\setbeamerfont{footnote}{size=\scriptsize}
